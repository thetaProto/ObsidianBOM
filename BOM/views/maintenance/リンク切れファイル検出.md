const allPaths = new Set(dv.pages().file.path);

const isMdLink = (path) => path && path.toLowerCase().endsWith(".md");

const broken = dv.pages("")
  .map(p => {
    const missing = (p.file.outlinks || [])
      .filter(link => isMdLink(link.path) && !allPaths.has(link.path));
    return missing.length ? { file: p.file, links: missing } : null;
  })
  .filter(Boolean);

dv.table(
  ["ファイル", "壊れたリンク数", "存在しないリンク先"],
  broken.map(({ file, links }) => [
    file.link,
    links.length,
    links.map(l => l.path).join(" / ")
  ])
);
